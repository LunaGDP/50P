    <button onclick="buscarPoema()">Leer</button>

<script>
    // Este es tu "archivo" de poemas
    const poemas = {
        1: "Pueblo mío, \nque por las mañanas despiertas \ncon el aroma del café y el rocío...",
        2: "Caminante, no hay camino, \nse hace camino al andar. \nAl andar se hace el camino...",
        15: "Este es el poema número 15, \nllegaste justo a tiempo \npara ver el sol caer.",
        // Puedes seguir agregando hasta el 50
    };

    function buscarPoema() {
        const num = document.getElementById('numeroPoema').value;
        const visor = document.getElementById('resultado');
        
        if (poemas[num]) {
            visor.innerHTML = `<h3>Poema Nro. ${num}</h3> ${poemas[num]}`;
        } else {
            visor.innerHTML = "<p style='color:red;'>Aún no hay un poema para este número o está fuera de rango.</p>";
        }
    }
</script>
